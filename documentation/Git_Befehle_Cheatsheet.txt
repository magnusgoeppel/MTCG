#################### Git Befehle Cheatsheet ####################

# Grundlegende Git-Befehle
git init                    	; Initialisiert ein neues Git-Repository
git clone [url]             	; Klont ein Repository
git remote add origin [url] 	; Verbindet mit einem Remote-Repository
git status                  	; Zeigt den Status von Änderungen
git add [Dateiname]         	; Markiert Änderungen in der Datei zur Aufnahme im nächsten Commit
git commit -m '[Nachricht]' 	; Speichert festgelegte Änderungen
git log                     	; Zeigt eine Liste aller Commits
git push -u origin [Branchname]	; Schiebt Ihre Commits zu GitHub
git pull                    	; Holt Änderungen von einem Remote-Repository

# Arbeiten mit Branches
git branch                 		; Listet alle lokalen Branches auf
git branch [Branchname]     	; Erstellt einen neuen Branch
git checkout [Branchname]   	; Wechselt zu einem Branch
git merge [Branchname]      	; Kombiniert die Branch-Historie
git branch -d [Branchname]  	; Löscht einen Branch
